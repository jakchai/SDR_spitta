Rebuild LibIIO
cd /usr/local/src
git clone https://github.com/analogdevicesinc/libiio.git
cd /usr/local/src/libiio/
git clean -d -f -x
cmake .
make
sudo make install

create service 
=========
vi /etc/systemd/system/iiod.service

[Service]
ExecStart=/usr/sbin/iiod

save > exit

การใช้งาน service 
systemctl enable iiod //เปิดใช้งานจะทำงานทุกครั้งเมื่อเปิดเครื่อง
service disable iiod //หยุดการทำงานเมือรีสตาร์ทเครื่องจะไม่ทำงาน
systemctl start iiod 
systemctl stop iiod
systemctl restart iiod
systemctl ststus iiod //ดู สถานะการทำงาน

ln -s /etc/init/iiod.conf /etc/iio.d/iiod.conf
